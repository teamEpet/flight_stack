100  bool set_IP4_address(const std::string& in_address)
101  {
102  std::stringstream ss(in_address);
103  int a,b,c,d; //to store the 4 ints
104  char ch; //to temporarily store the '.'
105  ss >> a >> ch >> b >> ch >> c >> ch >> d;
106  LOCATOR_ADDRESS_INVALID(address);
107  address[12] = (octet)a;
108  address[13] = (octet)b;
109  address[14] = (octet)c;
110  address[15] = (octet)d;
111  return true;
112  }
