136 class BinaryPropertyHelper
137 {
138  public:
139 
140  static size_t serialized_size(const BinaryProperty& binary_property, size_t current_alignment = 0)
141  {
142  if(binary_property.propagate())
143  {
144  size_t initial_alignment = current_alignment;
145 
146  current_alignment += 4 + alignment(current_alignment, 4) + binary_property.name().size() + 1;
147  current_alignment += 4 + alignment(current_alignment, 4) + binary_property.value().size();
148 
149  return current_alignment - initial_alignment;
150  }
151  else
152  return 0;
153  }
154 
155  static size_t serialized_size(const BinaryPropertySeq& binary_properties, size_t current_alignment = 0)
156  {
157  size_t initial_alignment = current_alignment;
158 
159  current_alignment += 4 + alignment(current_alignment, 4);
160  for(auto binary_property = binary_properties.begin(); binary_property != binary_properties.end(); ++binary_property)
161  current_alignment += serialized_size(*binary_property, current_alignment);
162 
163  return current_alignment - initial_alignment;
164  }
165 
166  private:
167 
168  inline static size_t alignment(size_t current_alignment, size_t dataSize) { return (dataSize - (current_alignment % dataSize)) & (dataSize-1);}
169 };
