134 class PropertyHelper

138  static size_t serialized_size(const Property& property, size_t current_alignment = 0)
139  {
140  if(property.propagate())
141  {
142  size_t initial_alignment = current_alignment;
143 
144  current_alignment += 4 + alignment(current_alignment, 4) + property.name().size() + 1;
145  current_alignment += 4 + alignment(current_alignment, 4) + property.value().size() + 1;
146 
147  return current_alignment - initial_alignment;
148  }
149  else
150  return 0;
151  }
152 
153  static size_t serialized_size(const PropertySeq& properties, size_t current_alignment = 0)
154  {
155  size_t initial_alignment = current_alignment;
156 
157  current_alignment += 4 + alignment(current_alignment, 4);
158  for(auto property = properties.begin(); property != properties.end(); ++property)
159  current_alignment += serialized_size(*property, current_alignment);
160 
161  return current_alignment - initial_alignment;
162  }
163 
164  private:
165 
166  inline static size_t alignment(size_t current_alignment, size_t dataSize) { return (dataSize - (current_alignment % dataSize)) & (dataSize-1);}
167 };
